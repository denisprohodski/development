pipeline {
    agent any

    stages {
        stage('Checkout') {
            steps {
                checkout scm
            }
        }

        stage('Build and Run Python Script') {
            steps {
                script {
                    sh 'pip install -r requirements.txt'
                }

                script {
                    sh 'python test_cases.py'
                }
            }
        }
    }
}
